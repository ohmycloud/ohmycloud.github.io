2016-11-10

遍历日期

#!/bin/bash                                                                                                                        
beg_date=`date -d "$1" +%s`                                                                                                        
end_date=`date -d "$2" +%s`                                                                                                        
                                                                                                                                   
echo ${beg_date}, ${end_date}                                                                                                      
                                                                                                                                   
for (( i=${beg_date};i<=${end_date};i=i+86400))                                                                                    
do                                                                                                                                 
        day=`date -d @${i} "+%Y-%m-%d"`                                                                                            
        echo ${day}                                                                                                                
       ./diyiqi.sh $3 $4 $5 -d ${day}                                                                                              
       sleep 360
done
